  class Conformance extends Resource         -- A conformance statement returned by request in an RESTful framework
  {
    attribute id : id;                       -- Globally unique Conformance Statement Id (Must be a UUID)
    attribute date : dateTime;               -- Date that the conformance statement is published
    attribute publisher : Conformance_publisher; -- The organization that publishes this conformance statement
    attribute software : Conformance_software; -- The software that is covered by this conformance statement
    attribute mode : code;                   -- client | server
    attribute resource : Constraint;         -- Resource Type with constraints
    attribute operation : Conformance_operation;
    attribute extensions : Extensions;       -- See Extensions
    attribute text : Narrative;              -- Text summary of conformance profile for human interpretation
  }

  class Conformance_software                 -- The software that is covered by this conformance statement
  {
    attribute name : String;                 -- Name software is known by
    attribute version : String;              -- Version covered by this statement
    attribute releaseDate : dateTime;        -- Date this version released
  }

  class Conformance_operation
  {
    attribute read : boolean;                -- if supported
    attribute vread : boolean;               -- if supported
    attribute update : boolean;              -- if supported
    attribute delete : boolean;              -- if supported
    attribute validate : boolean;            -- if supported
    attribute history : boolean;             -- if supported
    attribute transaction : Conformance_operation_transaction; -- only if supported
    attribute search : Conformance_operation_search; -- only if supported
    attribute create : Conformance_operation_create; -- if supported
    attribute updates : Conformance_operation_updates; -- if supported
    attribute schema : boolean;              -- if supported
  }

  class Conformance_operation_updates extends boolean -- if supported
  {
    attribute param : String;                -- update filter params supported
  }

  class Conformance_operation_search         -- only if supported
  {
    attribute param : String;                -- search params supported
  }

  class Conformance_operation_create         -- if supported
  {
    attribute id : code;                     -- source of id: client | server | either
  }

  class Conformance_publisher                -- The organization that publishes this conformance statement
  {
    attribute name : String;                 -- Name of Organization
    attribute address : Address;             -- Address of Organization
    attribute contact : Contact;             -- Contacts for Organization
  }

  class Conformance_operation_transaction    -- only if supported
  {
    attribute name : code;                   -- transaction names supported
  }

  class Document extends Resource            -- A documentation of clinical observations and services that are aggregated together into a single statement of clinical meaning that establishes it's own context. A clinical document is composed of a set of resources that include both human and computer readable portions. A human must attest to the accuracy of the human readable portion, and may authenticate and/or sign the entire whole
  {
    attribute id : id;                       -- the unique id of a clinical document
    attribute instant : instant;             -- the document creation time, when the document first came into being. Where the CDA document is a transform from an original document in some other format, the ClinicalDocument.effectiveTime is the time the original document is created.
    attribute type : CodeableConcept;        -- specifying the particular kind of document (e.g. History and Physical, Discharge Summary, Progress Note)
    attribute title : String;                -- the title of the document
    attribute setId : id;                    -- Represents an identifier that is common across all document revisions
    attribute version : integer;             -- used to version successive replacement documents
    attribute replaces : id;                 -- If this document replaces another
    attribute subject : Resource;            -- who the document is about
    attribute author : Document_author;      -- Author (contributed content to document)
    attribute attestor : Document_attestor;  -- a participant who has attested to the accuracy of the document
    attribute recipient : Resource;          -- expected to receive a copy 
    attribute custodian : Resource;          -- org which maintains the document.
    attribute event : Resource;              -- the main Act, such as a colonoscopy or an appendectomy, being documented
    attribute encounter : Resource;          -- context of the document
    attribute section : Document_section;    -- Document is broken into sections
    attribute extensions : Extensions;       -- See Extensions
    attribute text : Narrative;              --  
  }

  class Document_attestor                    -- a participant who has attested to the accuracy of the document
  {
    attribute mode : code;                   -- The type of attestation the authenticator offers
    attribute time : dateTime;               -- When document attested
    attribute party : Resource;              -- who attested the document
  }

  class Document_author                      -- Author (contributed content to document)
  {
    attribute time : dateTime;               -- When authoring happened
    attribute party : Resource;              -- who/what authored the final document
  }

  class Document_section                     -- Document is broken into sections
  {
    attribute type : CodeableConcept;        -- type of section (recommended)
    attribute instant : instant;             -- the section creation time (sections are often re-used in several documents).
    attribute author : Document_section_author; -- if section author different to document
    attribute enterer : Resource;            -- The person or device that performed the data entry leading to this section. Where there is more than one candidate, pick the most proximal to the message. Can provide other enterers in extensions
    attribute subject : Resource;            -- if section different to document
    attribute informant : Resource;          -- provided information in section
    attribute content : Resource;            -- the actual content of the section
    attribute section : Document_Section;    -- nested Section
  }

  class Document_section_author              -- if section author different to document
  {
    attribute time : dateTime;               -- When authoring happened
    attribute party : Resource;              -- who/what authored the section
  }

  class Message extends Resource             -- A message that contains FHIR resources
  {
    attribute id : id;                       -- Unique Id for this message
    attribute threadId : id;                 -- Id of the thread - a series of messages that pertain to the same logical sequence, and are all identified by the same thread identifier
    attribute instant : instant;             -- Instant the message was sent
    attribute event : code;                  -- Code that identifies the event this message represents, and connects it with the event definition in the FHIR specification
    attribute response : Message_response;   -- Information about the the message that this message is a response to - if it is a response
    attribute source : Resource;             -- The source application from which this message originated
    attribute destination : Resource;        -- The destination application which the message is intended for
    attribute enterer : Resource;            -- The person or device that performd the data entry leading to this message. Where there is more than one candidate, pick the most proximal to the message. Can provide other enterers in extensions
    attribute author : Resource;             -- The logical author of the message - the person or device that decided it should happen. Where there is more than one candidate, pick the most proximal to the message. Can provide other enterers in extensions
    attribute responsible : Resource;        -- The person or organization that accepts overall responsbility for the contents of the message. The implication is that the message event happened under the policies of the responsible party
    attribute effective : Interval;          -- The effective time - the real world time of the even that the message represents. Usually this is just a starting time, but some message events also have an end time (do x for period y)
    attribute reason : CodeableConcept;      -- The cause of the event - a reason for why this message is being sent
    attribute data : Resource;               -- The actual data of the message - a reference to the focus class of the message. 
    attribute extensions : Extensions;       -- See Extensions
    attribute text : Narrative;              --  
  }

  class Message_response                     -- Information about the the message that this message is a response to - if it is a response
  {
    attribute id : id;                       -- The id of the message that this a response to
    attribute code : code;                   -- Code that identifies the type of response to the message - whether it was successful or not, and whether it should be resent or not
    attribute duplicate : boolean;           -- True if this is not the first response, because the request message has been received more than once
  }

  class Animal extends Resource              -- An animal that has relevance to the care process -usually this is for animals that are patients.

  class Prescription extends Resource        -- Directions provided by a prescribing practitioner for a specific medication to be administered to an individual
  {
    attribute id : id;                       -- Master Resource Id, always first in all resources
    attribute identifier : HumanId;          -- A identifier used in an external system and associated with this medication
    attribute status : code;                 -- Actual status of the prescription
    attribute patient : Resource;            -- The patient the prescription is prescribing medicine for
    attribute prescriber : Resource;         -- The clinician or doctor prescribing the medication
    attribute prescribed : dateTime;         -- Date/time on which the prescription was written
    attribute dispense : Prescription_dispense; -- Details of the dispense as requested by the prescriber
    attribute medicine : Prescription_medicine; -- The one single medicatine, vaccine or therapeutic good prescribed in this prescription.
    attribute administrationRequest : Prescription_administrationRequest; -- Instructions for the use of the medication. Includes details about the timing schedule, dosis amounts and additional usage instructions.
    attribute reason : CodeableConcept;      -- Diagnosis which is the reason for prescribing this medicine
    attribute extensions : Extensions;       -- See Extensions
    attribute text : Narrative;              -- Text summary of the prescription, for human interpretation
  }

  class Prescription_medicine                -- The one single medicatine, vaccine or therapeutic good prescribed in this prescription.
  {
    attribute productCode : Coding;          -- Coded representation of medicine
    attribute description : String;          -- Textual description of medicine, including strength and ingredients
    attribute activeIngredient : Prescription_medicine_activeIngredient; -- The substance in the medication formulation that is pharmaceutically active and is responsible for the medication's therapeutic effect
    attribute inactiveIngredient : Prescription_medicine_inactiveIngredient; -- Ingredients in the medication that are not active
  }

  class Prescription_medicine_activeIngredient -- The substance in the medication formulation that is pharmaceutically active and is responsible for the medication's therapeutic effect
  {
    attribute productCode : Coding;          -- Coded representation of active ingredient
    attribute quantity : Ratio;              -- Quantity of active ingredient expressed in relation to the whole of the prepared medicine
  }

  class Prescription_medicine_inactiveIngredient -- Ingredients in the medication that are not active
  {
    attribute productCode : Coding;          -- Coded representation of the inactive ingredient
    attribute quantity : Ratio;              -- Quantity of inactive ingredient expressed in relation to the whole of the prepared medicine
  }

  class Prescription_dispense                -- Details of the dispense as requested by the prescriber
  {
    attribute repeats : integer;             -- Requested number of repeats
    attribute quantity : Quantity;           -- Requested quantity per repeat
    attribute dispenser : Resource;          -- Person to fullfil the requested dispense
  }

  class Prescription_administrationRequest   -- Instructions for the use of the medication. Includes details about the timing schedule, dosis amounts and additional usage instructions.
  {
    attribute description : String;          -- Textual description of the use of the medication.
    attribute totalPeriodicDosis : Ratio;    -- Total dose per day/week or other period when more specific information is missing or cannot be expressed using the timing specifications.
    attribute start : dateTime;              -- First moment on which medication should be taken
    attribute end : dateTime;                -- Last moment on which medication should be taken
    attribute duration : Quantity;           -- Total duration of administration
    attribute numberOfAdministrations : integer; -- Maximum number of separate administrations before the instruction ends.
    attribute dosageInstruction : Prescription_administrationRequest_dosageInstruction; -- Specification of dose and schedule for administration
  }

  class Prescription_administrationRequest_dosageInstruction -- Specification of dose and schedule for administration
  {
    attribute precondition : CodeableConcept; -- Precondition for starting the administration specified in this instruction
    attribute prn : code;                    -- Pro re nate, "If necessary": Specifies whether administration depens on the state and symptoms of the patient
    attribute additionalInstruction : CodeableConcept; -- Additional details to guide administration. Especially relevant for medicine administered by patient
    attribute route : CodeableConcept;       -- Route of administration (oral, nasal, intravenous)
    attribute dose : ?;                      -- Dose per administration, expressed in units of the (prepared) product
    attribute rate : Quantity;               -- Flow-rate for IV
    attribute schedule : Schedule;           -- Schedule for administration. If multiple are given, they are considered to be active in parrallel
  }

  class LabReport extends Resource           -- The findings and interpretation of pathology tests performed on tissues and body fluids. This is typically done in a laboratory but may be done in other environments such as at the point of care
  {
    attribute id : id;                       -- Master Resource Id, always first in all resources
    attribute status : code;                 -- The status of the pathology test result as a whole
    attribute issued : instant;              -- The date and/or time that the result was issued from the source for the recorded ‘Test result status
    attribute patient : Resource;            -- The patient about who the report is about
    attribute admission : Resource;          -- The admission that this diagnostic investigation is associated with
    attribute laboratory : Resource;         -- The laboratory service that issued the report
    attribute reportId : id;                 -- The local ID assigned to the report by the order filler, usually by the Laboratory Information System (LIS).
    attribute requestDetail : LabReport_requestDetail; -- Details concerning a single pathology test requested.
    attribute reportName : CodeableConcept;  -- Identification of the pathology test performed, sometimes including specimen type.
    attribute service : CodeableConcept;     -- The diagnostic service that performs the examination e.g. biochemistry, haematology.
    attribute diagnosticTime : dateTime;     -- The diagnostically relevant time for this report
    attribute specimen : Resource;           -- Details about the specimen if all individual test results are derived from the same specimen
    attribute resultGroup : LabReport_resultGroup; -- A group of results. Results may be grouped by specimen, or by some value in LabReport.resultGroup.name to describe what binds all the results together.
    attribute conclusion : Narrative;        -- Concise and clinically contextualised narrative interpretation of the pathology test results.
    attribute codedDiagnosis : CodeableConcept; -- Codes for the conclusion
    attribute representation : Attachment;   -- Rich text representation of the entire result as issued by the diagnostic service. Multiple formats are allowed but they must be semantically equivalent.
    attribute extensions : Extensions;       -- See Extensions
    attribute text : Narrative;              -- Text summary of report, for human interpretation
  }

  class LabReport_resultGroup                -- A group of results. Results may be grouped by specimen, or by some value in LabReport.resultGroup.name to describe what binds all the results together.
  {
    attribute name : CodeableConcept;        -- A code or name that describes the group of results
    attribute specimen : Resource;           -- Details about the individual specimen to which these ‘Result group’ test results refer, where testing of multiple specimens is required.
    attribute result : LabReport_resultGroup_result; -- Specific detailed result, including both the value of the result item, and additional information that may be useful for clinical interpretation. Results include whatever specific data items pathology labs report as part of the clinical service; it is not confined to measurements.
  }

  class LabReport_resultGroup_result         -- Specific detailed result, including both the value of the result item, and additional information that may be useful for clinical interpretation. Results include whatever specific data items pathology labs report as part of the clinical service; it is not confined to measurements.
  {
    attribute name : CodeableConcept;        -- Identifies the meaning of the value
    attribute value : ?;                     -- Actual value of the result. Most result values will be numerical measurements, but others may be coded concepts, free text, or multimedia images
    attribute flag : code;                   -- Flag indicating the abnormal status of the result
    attribute status : code;                 -- The status of the result value
    attribute comments : String;             -- May include statements about significant, unexpected or unreliable. values, or information about the source of the value where this may be relevant to the interpretation of the result.
    attribute referenceRange : LabReport_resultGroup_result_referenceRange; -- Guidance on how to interpret the value by comparison to a normal or recommended range
  }

  class LabReport_resultGroup_result_referenceRange -- Guidance on how to interpret the value by comparison to a normal or recommended range
  {
    attribute meaning : CodeableConcept;     -- Code for the meaning of the reference range
    attribute range : ?;                     -- Actual value of the reference range.  May be a quantity (<20mg/L), an interval (10-20 umol/L), or some text
  }

  class LabReport_requestDetail              -- Details concerning a single pathology test requested.
  {
    attribute requestOrderId : Identifier;   -- The local ID assigned to the order by the order requester.
    attribute receiverOrderId : Identifier;  -- The local ID assigned to the test order by the order filler, usually by the Laboratory Information System (LIS).
    attribute requestTest : CodeableConcept; -- Identification of pathology test requested,
    attribute requester : Resource;          -- Details of the clinician or organisation requesting the laboratory test.
    attribute clinicalInfo : Resource;       -- Details of the clinical information provided to the laboratory along with the original request
  }
